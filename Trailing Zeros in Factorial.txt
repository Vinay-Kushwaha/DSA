public class Solution {
    
    public int trailingZeroes(int A) {
        int n =A;
        
        int result =1;
        int temp[] = { 2,5};
        int min = 1000000000;
        for(int i =0;i<2;i++ ){
            int p = temp[i];
            int exp =0;
            while(p<=n){
                exp = exp+(n/p);
                p=p*temp[i];
            }
            min= Math.min(min,exp);
        }
        return min;
    }
}
